\begin{denotation}[4.8cm]
  \item[ANN] Approximate Nearest Neighbor，近似最近邻
  \item[AUC] Area Under Curve，曲线下面积
  \item[BTO] Basic Tensor Operations，基础张量算子
  \item[CPU] Central Processing Unit，中央处理器
  \item[DMA] Direct Memory Access，直接内存访问
  \item[FO] Functional Operations，功能算子
  \item[GDS] GPUDirect Storage，GPU 直连存储访问机制
  \item[GPU] Graphics Processing Unit，图形处理器
  \item[H2D / D2H] Host-to-Device / Device-to-Host，主机到设备 / 设备到主机传输
  \item[I/O] Input/Output，输入/输出
  \item[KNN] K-Nearest Neighbors，$k$ 近邻
  \item[NIC] Network Interface Card，网络接口卡
  \item[NUMA] Non-Uniform Memory Access，非一致内存访问
  \item[PCIe] Peripheral Component Interconnect Express，外设组件高速互连
  \item[PR-AUC] Precision-Recall Area Under Curve，精确率—召回率曲线下面积
  \item[QPS] Queries Per Second，每秒查询数
  \item[RMM] RAPIDS Memory Manager，RAPIDS 内存管理器
  \item[ROC-AUC] Receiver Operating Characteristic Area Under Curve，受试者工作特征曲线下面积
  \item[SSD] Solid-State Drive，固态硬盘
  \item[Top-$k$ overlap] 近似排序结果与基线排序结果的前 $k$ 重合度
  \item[\texttt{adaptive\_ratio}] 多 GPU 动态分流比例策略
  \item[bounded batching] 在时延约束下受控执行的微批处理策略
  \item[rerank] 对粗排候选执行更高精度精排计算的步骤
\end{denotation}
